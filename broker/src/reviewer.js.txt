#!/usr/bin/env node
// Served by the broker at GET /reviewer.mjs; the box fetches + runs it as its exec command.
// Clones the PR, runs Claude as a full agent inside this throwaway microVM, extracts the
// findings JSON, and POSTs it to the broker /publish with the one-shot job token. Pure Node,
// self-contained — no stdout contract, the box exits after the callback.
import { execFileSync, spawnSync } from 'node:child_process'
import { mkdtempSync } from 'node:fs'
import { tmpdir } from 'node:os'
import path from 'node:path'

// The Worker fires this review and never attaches, so the exec is an "orphaned session" the
// runner SIGHUPs after ~5 min idle. A real review runs longer than that — ignore SIGHUP so we
// survive the reap and still POST the findings. (detach:true keeps the box itself alive.)
process.on('SIGHUP', () => {})

const E = process.env
const BASE = `origin/${E.BASE_REF}`

const PROMPT = `You are reviewing a pull request. The repo is checked out at the PR head; the base branch is the remote-tracking ref ${BASE}. You are inside a throwaway, isolated BoxLite microVM — it is safe to install anything and run the code.

1. Run: git diff --numstat ${BASE}...HEAD  and  git diff ${BASE}...HEAD, then read the diff and any surrounding code for context.
2. Judge only the changed code: correctness bugs, concurrency/lifecycle issues, security problems, broken API contracts, missing or weakened tests. Ignore style, formatting, naming.
3. Confirm with evidence — do not just guess. Where it helps, install deps and run the test suite, or write a short script to reproduce a suspected bug. You have a full shell (sudo, network). Put what you ran and the result in the finding body (e.g. ran npm test -> 2 failing in auth.test.ts).

Output ONLY a JSON object — no prose, no code fence — minimum words, worst issue first:
{"verdict":"looks good | N issues","changeMap":[{"file":"path","symbol":"function/class or null","loc":"+12/-3","note":"<=6 words"}],"findings":[{"path":"path","line":0,"endLine":null,"severity":"blocker|warning|nit","category":"correctness|security|concurrency|api|tests|other","title":"<=8 words","body":"one line: what breaks and when, cite a repro if you ran one","suggestion":"corrected code for exactly lines line..endLine, or null"}]}

Rules: changeMap = one row per meaningful changed symbol (loc from numstat). findings = only real problems; prefer a bug you REPRODUCED and put the command+result in body; line/endLine must be lines that exist in the head file; suggestion replaces exactly those lines or is null (never partial). No praise, no narration. Nothing wrong => verdict "looks good", findings [].

PR #${E.PR} of ${E.REPO}. Base: ${BASE}. Head: ${(E.HEAD_SHA || '').slice(0, 7)}.`

function extractJson(text) {
  const raw = String(text)
  const fence = raw.match(/```(?:json)?\s*\n?([\s\S]*?)```/i)
  const scope = fence ? fence[1] : raw
  const s = scope.indexOf('{')
  const e = scope.lastIndexOf('}')
  if (s === -1 || e <= s) throw new Error('no JSON object in model output')
  return JSON.parse(scope.slice(s, e + 1))
}

async function publish(payload) {
  try {
    await fetch(`${E.BROKER}/publish`, {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ boxId: E.BOX_ID, job: E.JOB_TOKEN, ...payload }),
    })
  } catch {
    /* box exits regardless; the broker's reaper tears it down if the callback never lands */
  }
}

async function main() {
  try {
    try {
      execFileSync('claude', ['--version'], { stdio: 'ignore' })
    } catch {
      execFileSync('sudo', ['npm', 'install', '-g', '@anthropic-ai/claude-code'], { stdio: ['ignore', 2, 2] })
    }

    const git = (a) => execFileSync('git', a, { stdio: ['ignore', 'pipe', 'inherit'], encoding: 'utf8' })
    const repoDir = path.join(mkdtempSync(path.join(tmpdir(), 'pr-review-')), 'repo')
    // contents:read clone token — used only for the clone, then scrubbed before Claude runs.
    git(['clone', '-q', '--depth', '200', `https://x-access-token:${E.GH_CLONE_TOKEN}@github.com/${E.REPO}.git`, repoDir])
    git(['-C', repoDir, 'fetch', '-q', '--depth', '200', 'origin', `+refs/pull/${E.PR}/head`, `+refs/heads/${E.BASE_REF}:refs/remotes/origin/${E.BASE_REF}`])
    git(['-C', repoDir, 'checkout', '-q', 'FETCH_HEAD'])
    git(['-C', repoDir, 'remote', 'set-url', 'origin', `https://github.com/${E.REPO}.git`])

    const args = ['-p', PROMPT, '--allowedTools', 'Read,Grep,Glob,Write,Bash', '--output-format', 'json', '--max-turns', '80']
    if (E.MODEL) args.push('--model', E.MODEL)
    const childEnv = { ...process.env }
    delete childEnv.GH_CLONE_TOKEN // the model never sees the git or callback tokens
    delete childEnv.JOB_TOKEN
    const res = spawnSync('claude', args, {
      cwd: repoDir,
      env: childEnv,
      encoding: 'utf8',
      maxBuffer: 64 * 1024 * 1024,
      stdio: ['ignore', 'pipe', 'pipe'],
    })
    if (res.status !== 0) {
      await publish({ error: `claude exited ${res.status}: ${String(res.stderr || '').slice(0, 400)}` })
      return
    }
    let findings
    try {
      findings = extractJson(JSON.parse(res.stdout).result)
    } catch (e) {
      await publish({ error: `bad review output: ${e.message}` })
      return
    }
    await publish({ findings })
  } catch (e) {
    await publish({ error: String((e && e.message) || e).slice(0, 400) })
  }
}

main()
